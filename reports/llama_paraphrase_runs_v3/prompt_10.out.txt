import statistics
from typing import List, Tuple

def task_func(pairs: List[Tuple[str, str]]) -> Tuple[float, float, int]:
    # Convert string pairs to integers
    numbers = [int(pair[0]) + int(pair[1]) for pair in pairs]
    
    # Sum all numbers
    total_sum = sum(numbers)
    
    # Generate a random list of that length with ints in [0, 9]
    random_list = [statistics.randint(0, 9) for _ in range(total_sum)]
    
    # Calculate mean, median, mode
    mean = statistics.mean(random_list)
    median = statistics.median(random_list)
    mode = statistics.mode(random_list)
    
    return mean, median, mode
