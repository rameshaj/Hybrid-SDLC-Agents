import os
import shutil

def task_func(log_dir: str, archive_path: str) -> str:
    # Create a tar.gz archive of all .log files in log_dir
    shutil.make_archive(archive_path, 'gztar', log_dir)
    
    # Delete the originals
    for file in os.listdir(log_dir):
        if file.endswith('.log'):
            os.remove(os.path.join(log_dir, file))
    
    return archive_path
