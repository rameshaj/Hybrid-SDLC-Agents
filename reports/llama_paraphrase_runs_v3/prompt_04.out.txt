import numpy as np

def task_func(letters: list[str], k: int) -> dict[str, float]:
    result = {}
    for letter in letters:
        random_numbers = np.random.randint(0, 51, k)
        mean_value = np.mean(random_numbers)
        result[letter] = mean_value
    return result
